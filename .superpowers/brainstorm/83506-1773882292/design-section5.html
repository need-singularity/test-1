5-4. 구성요소 인터페이스 (공통 프로토콜)

class Component(Protocol): """모든 15개 구성요소가 구현하는 공통 인터페이스""" name: str # 예: "ricci_flow" layer: str # 예: "reasoning" def execute(self, state: TopologyState) -> TopologyState: """입력 상태를 받아 변환된 상태를 반환""" ... def measure(self, state: TopologyState) -> dict[str, float]: """현재 상태의 지표(β,χ,r,Φ,λ) 측정""" ... def cost(self) -> float: """이 구성요소의 상대적 계산 비용""" ... @dataclass class TopologyState: """구성요소 간 전달되는 공통 상태""" complex: Any # SimplexTree | graph | hypergraph curvature: np.ndarray # 곡률 텐서 (이산 근사) metrics: dict # 현재 지표 history: list # 변형 경로 기록 metadata: dict # 추가 정보
